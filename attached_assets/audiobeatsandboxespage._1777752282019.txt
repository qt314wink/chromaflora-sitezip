const BeatsAndBoxesPage = ({ setPage }) => {
 const [isPlaying, setIsPlaying] = useState(false);
 const [progress, setProgress] = useState(0);
 const audioRef = useRef(null);

 // Handle actual audio play/pause
 const togglePlay = () => {
   if (!audioRef.current) return;
   
   if (isPlaying) {
     audioRef.current.pause();
   } else {
     audioRef.current.play();
   }
   setIsPlaying(!isPlaying);
 };

 // Update progress bar dynamically
 const handleTimeUpdate = () => {
   if (!audioRef.current) return;
   const current = audioRef.current.currentTime;
   const duration = audioRef.current.duration;
   if (duration > 0) {
     setProgress((current / duration) * 100);
   }
 };

 const handleAudioEnded = () => {
   setIsPlaying(false);
   setProgress(0);
 };

 return (
   <div className="pt-24 md:pt-32 pb-20 min-h-screen bg-white animate-in slide-in-from-right duration-500">
     <div className="max-w-4xl mx-auto px-4 mb-8">
       <button onClick={() => setPage('projects')} className="flex items-center gap-2 text-xs font-black uppercase tracking-widest hover:text-[#23a6d5] transition-colors">
         <ArrowLeft size={14} /> Back to Workspaces
       </button>
     </div>

     <header className="max-w-4xl mx-auto px-4 mb-12 relative">
       <span className="inline-block bg-[#23a6d5] text-white px-3 py-1 text-[10px] font-black uppercase tracking-widest mb-6">Case Study // Civic Infrastructure</span>
       <h1 className="kinetic-text text-6xl md:text-8xl leading-none mb-6">The Bassline &<br/>The Breadline</h1>
       <p className="text-xl md:text-2xl font-medium text-gray-500 leading-relaxed max-w-2xl">
         How a Philadelphia dance party became a prototype for urban mutual aid and turned art into infrastructure.
       </p>
     </header>

     <div className="max-w-6xl mx-auto px-4 mb-16">
       <div className="w-full h-[400px] brutalist-border overflow-hidden bg-black">
         <img 
           src={BEATS_BOXES_IMG} 
           alt="Beats and Boxes Hero"
           className="w-full h-full object-cover opacity-80 hover:opacity-100 transition-opacity duration-700 hover:scale-105 transform"
         />
       </div>
     </div>

     <div className="max-w-4xl mx-auto px-4 mb-16">
       {/* REAL AUDIO ELEMENT (Hidden) */}
       <audio 
         ref={audioRef}
         src="Beats__Boxes_How_a_Philadelphia_Dance_Party_Became_a_Prototype_for_Urban_Mutual_Aid_and_Turned_Art_into_Infrastructure.mp3"
         onTimeUpdate={handleTimeUpdate}
         onEnded={handleAudioEnded}
         preload="metadata"
       />

       <div className="brutalist-border bg-black text-white p-6 md:p-8 flex flex-col md:flex-row items-center gap-6 group">
         <button onClick={togglePlay} className="shrink-0 text-[#23a6d5] hover:text-white transition-colors">
           {isPlaying ? <PauseCircle size={64} /> : <PlayCircle size={64} />}
         </button>
         <div className="flex-1 w-full">
           <p className="text-[10px] font-black uppercase tracking-widest text-gray-400 mb-2">Audio Documentary • Co. La'Bright</p>
           <h3 className="text-xl md:text-2xl font-black uppercase mb-4">Beats & Boxes: Field Notes</h3>
           <div className="h-2 w-full bg-gray-800 relative cursor-pointer" onClick={(e) => {
             // Allow seeking by clicking the progress bar
             if(audioRef.current) {
               const rect = e.currentTarget.getBoundingClientRect();
               const clickX = e.clientX - rect.left;
               const newTime = (clickX / rect.width) * audioRef.current.duration;
               audioRef.current.currentTime = newTime;
             }
           }}>
              <div 
                 className="absolute top-0 left-0 h-full bg-[#23a6d5] transition-all duration-100 ease-linear pointer-events-none" 
                 style={{ width: `${progress}%` }} 
              />
           </div>
         </div>
       </div>
     </div>

     <section className="bg-gray-50 py-16 border-y-4 border-black">
       <div className="max-w-4xl mx-auto px-4 grid grid-cols-1 md:grid-cols-12 gap-12">
          <div className="md:col-span-4">
            <h2 className="text-3xl font-black uppercase sticky top-32">Art Feeds<br/>The City</h2>
          </div>
          <div className="md:col-span-8 space-y-6 text-base md:text-lg font-medium leading-relaxed">
            <p className="first-letter:text-7xl first-letter:font-black first-letter:float-left first-letter:mr-3 first-letter:mt-2">
               There are two queues forming in Philadelphia. One winds under neon and mirrors toward a dance floor; the other threads past folding tables toward a pantry. They look like different stories—pleasure vs. need—but they share the same spine: <strong>people moving in rhythm to stay alive.</strong>
            </p>
            <p>
               A shutdown throttles federal lifelines. Families in 19135 are staring at an empty EBT balance and a rising PECO bill. We can wait for someone in a distant chamber to remember us, or we can do what cities have always done when empires stall: braid our own safety net from what we already have—our venues, our crews, our makers, our neighbors.
